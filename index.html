import random import os import time class FalloutClicker: def __init__(self): self.caps = 0 self.clicks = 0 self.multiplier = 1 # Passive Income Items self.upgrades = { "1": {"name": "Mutfruit Plant", "cost": 500, "gps": 10}, "2": {"name": "Water Purifier", "cost": 2500, "gps": 60}, "3": {"name": "Industrial Purifier", "cost": 12000, "gps": 300}, "4": {"name": "T-60 Power Armor", "cost": 50000, "gps": 1500}, } def get_loot(self): roll = random.randint(1, 1000) # Rarity Logic if roll == 1: return "Ultra Rare: Perfectly Preserved Pie", 100000 elif roll <= 10: return "Mythic: Alien Blaster", 10000 elif roll <= 50: return "Legendary: Deathclaw Hide", 1000 elif roll <= 200: return "Rare: Gold Fusion Core", 300 else: return "Common: Desk Fan", 100 def clear(self): os.system('cls' if os.name == 'nt' else 'clear') def start(self): while True: self.clear() print("--- ☢️ FALLOUT 4: CAPS CLICKER ☢️ ---") print(f"CAPS: {self.caps:,}") print(f"CLICKS: {self.clicks}") print("-" * 30) print("[ENTER] to Click | [S] Shop | [Q] Quit") inp = input("> ").lower() if inp == "": self.clicks += 1 self.caps += 1 # Every 10 Clicks: 10k Bonus + Random Loot if self.clicks % 10 == 0: self.caps += 10000 item_name, item_val = self.get_loot() self.caps += item_val print(f"\n⚡ 10 CLICK BONUS: +10,000 Caps!") print(f"🎁 LOOT FOUND: {item_name} (+{item_val:,} Caps)") time.sleep(1.2) # Every 20 Clicks: The 1M Bonus Item if self.clicks % 20 == 0: self.caps += 1000000 print(f"\n🔥 ULTRA SPECIAL: You found a 'Waste-Wand'!") print(f"💰 SOLD FOR: 1,000,000 Caps") time.sleep(1.5) elif inp == "s": self.clear() print("--- 🛒 DIAMOND CITY MARKET 🛒 ---") for k, v in self.upgrades.items(): print(f"{k}. {v['name']} | Cost: {v['cost']:,} | Income: +{v['gps']}/sec") buy = input("\nSelect item # to buy (or 'b' to go back): ") if buy in self.upgrades: item = self.upgrades[buy] if self.caps >= item['cost']: self.caps -= item['cost'] print(f"Successfully bought {item['name']}!") # Note: In a terminal app, passive income usually requires threading. # For this simple script, we'll give a huge click bonus instead! self.multiplier += (item['gps'] // 10) else: print("Not enough caps, smoothskin!") time.sleep(1) elif inp == "q": break if __name__ == "__main__": game = FalloutClicker() game.start()
