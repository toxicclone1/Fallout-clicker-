import random import os import time class FalloutClicker: def __init__(self): self.caps = 0 self.clicks = 0 self.defense = 0 self.raid_level = 1 # Shop: Name, Cost, Passive/Click Bonus self.shop = { "1": {"name": "Mutfruit Plant", "cost": 500, "bonus": 10}, "2": {"name": "Water Purifier", "cost": 2500, "bonus": 50}, "3": {"name": "T-60 Power Armor", "cost": 50000, "bonus": 1000}, } # Settlement: Name, Cost, Defense Points self.settlement_items = { "1": {"name": "Junk Wall", "cost": 1000, "def": 10}, "2": {"name": "Machine Gun Turret", "cost": 5000, "def": 50}, "3": {"name": "Heavy Laser Turret", "cost": 20000, "def": 250}, } def clear(self): os.system('cls' if os.name == 'nt' else 'clear') def trigger_raid(self): raid_strength = self.raid_level * 100 print(f"\n🚨 WARNING: Raiders are attacking! (Strength: {raid_strength})") time.sleep(1) if self.defense >= raid_strength: reward = self.raid_level * 5000 self.caps += reward print(f"✅ Defenses held! You looted the raiders for {reward:,} caps.") self.raid_level += 1 else: loss = int(self.caps * 0.2) self.caps -= loss print(f"❌ Defenses breached! Raiders stole {loss:,} caps.") time.sleep(2) def get_loot(self): roll = random.randint(1, 100) if roll == 1: return "Ultra Rare: Waste-Wand", 1000000 if roll <= 5: return "Mythic: Alien Blaster", 10000 if roll <= 15: return "Legendary: Deathclaw Hide", 1000 if roll <= 35: return "Rare: Gold Fusion Core", 300 return "Common: Desk Fan", 100 def start(self): while True: self.clear() print(f"--- ☢️ FALLOUT CLICKER | Defense: {self.defense} | Raid Level: {self.raid_level} ---") print(f"CAPS: {self.caps:,} | CLICKS: {self.clicks}") print("-" * 50) print("[ENTER] Click | [S] Shop | [H] Settlement (Defense) | [Q] Quit") choice = input("> ").lower() if choice == "": self.clicks += 1 self.caps += 1 if self.clicks % 10 == 0: self.caps += 10000 item, val = self.get_loot() self.caps += val print(f"\n🎁 10-Click Bonus! Found: {item} (+{val:,} caps)") if self.clicks % 20 == 0: print("🔥 SPECIAL: 1M Cap Item Bonus!") self.caps += 1000000 # Random chance for a raid every 10 clicks if random.random() > 0.7: self.trigger_raid() else: time.sleep(1) elif choice == "s": self.menu(self.shop, "DIAMOND CITY MARKET") elif choice == "h": self.menu(self.settlement_items, "SETTLEMENT WORKSHOP", is_defense=True) elif choice == "q": break def menu(self, source, title, is_defense=False): self.clear() print(f"--- {title} ---") for k, v in source.items(): stat_type = "DEF" if is_defense else "BONUS" stat_val = v['def'] if is_defense else v['bonus'] print(f"{k}. {v['name']} | Cost: {v['cost']:,} | {stat_type}: +{stat_val}") buy = input("\nBuy # or [B]ack: ") if buy in source: item = source[buy] if self.caps >= item['cost']: self.caps -= item['cost'] if is_defense: self.defense += item['def'] print(f"Purchased {item['name']}!") else: print("Not enough caps!") time.sleep(1) if __name__ == "__main__": game = FalloutClicker() game.start()
